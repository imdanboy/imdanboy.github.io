def name_to_title(names):
    name_to_title_dict = {
                        "Capt":       "Officer",
                        "Col":        "Officer",
                        "Major":      "Officer",
                        "Jonkheer":   "Royalty",
                        "Don":        "Royalty",
                        "Sir" :       "Royalty",
                        "Dr":         "Officer",
                        "Rev":        "Officer",
                        "the Countess":"Royalty",
                        "Dona":       "Royalty",
                        "Mme":        "Mrs",
                        "Mlle":       "Miss",
                        "Ms":         "Mrs",
                        "Mr" :        "Mr",
                        "Mrs" :       "Mrs",
                        "Miss" :      "Miss",
                        "Master" :    "Master",
                        "Lady" :      "Royalty"
            }
    
    names = names.map(lambda name: name.split(',')[1].split('.')[0].strip())
    titles = names.map(name_to_title_dict)
    return titles

title_encoder = LabelEncoder()
title_oh_encoder = OneHotEncoder(sparse=False)

titles = name_to_title(df_train['Name'])
encoded_titles = title_encoder.fit_transform(titles)
oh_encoded_titles = title_oh_encoder.fit_transform(encoded_titles[:, np.newaxis])
for i, title in enumerate(title_encoder.classes_):
    X_train['Title_' + title] = oh_encoded_titles[:, i]

titles = name_to_title(df_test['Name'])

    X_test['Title_' + title] = oh_encoded_titles[:, i]

def plot_validation_curve(train_scores, test_scores, x_lim):
    train_scores_mean = np.mean(train_scores, axis=1)
    test_scores_mean = np.mean(test_scores, axis=1)
    plt.figure(figsize=(12,8))
    plt.plot(x_lim, train_scores_mean, label='train')
    plt.plot(x_lim, test_scores_mean, label='test')
    plt.legend()
